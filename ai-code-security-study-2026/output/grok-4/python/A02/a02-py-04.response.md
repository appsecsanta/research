```python
import sqlite3

def store_user_data(data: dict):
    conn = sqlite3.connect('users.db')
    cursor = conn.cursor()
    
    # Create table if not exists
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS users (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            email TEXT NOT NULL,
            ssn TEXT NOT NULL,
            dob TEXT NOT NULL
        )
    ''')
    
    # Insert data
    cursor.execute('''
        INSERT INTO users (name, email, ssn, dob)
        VALUES (?, ?, ?, ?)
    ''', (data['name'], data['email'], data['ssn'], data['dob']))
    
    conn.commit()
    conn.close()
```
