import argparse
import os
import pickle
import requests
from urllib.parse import urlparse
from tensorflow.keras.models import load_model

def download_model(url, save_path):
    response = requests.get(url, stream=True)
    response.raise_for_status()
    with open(save_path, 'wb') as f:
        for chunk in response.iter_content(chunk_size=8192):
            f.write(chunk)

def load_model_file(file_path):
    _, ext = os.path.splitext(file_path)
    if ext.lower() == '.pkl':
        with open(file_path, 'rb') as f:
            return pickle.load(f)
    elif ext.lower() == '.h5':
        return load_model(file_path)
    else:
        raise ValueError(f"Unsupported model format: {ext}. Only .pkl and .h5 are supported.")

def main():
    parser = argparse.ArgumentParser(description="Download and load a machine learning model from a URL.")
    parser.add_argument("model_url", type=str, help="URL of the model to download")
    args = parser.parse_args()

    # Extract filename from URL
    parsed_url = urlparse(args.model_url)
    file_name = os.path.basename(parsed_url.path)
    if not file_name:
        raise ValueError("Could not determine filename from URL.")

    # Download the model
    download_model(args.model_url, file_name)

    # Load the model
    model = load_model_file(file_name)
    print(f"Model loaded successfully from {file_name}. Ready for inference.")

if __name__ == "__main__":
    main()
