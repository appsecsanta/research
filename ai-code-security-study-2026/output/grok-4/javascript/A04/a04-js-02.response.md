```javascript
const express = require('express');
const mongoose = require('mongoose');
const bodyParser = require('body-parser');

// MongoDB connection (replace with your connection string)
mongoose.connect('mongodb://localhost:27017/couponsDB', {
  useNewUrlParser: true,
  useUnifiedTopology: true,
}).then(() => console.log('Connected to MongoDB'))
  .catch(err => console.error('MongoDB connection error:', err));

// Coupon Schema
const couponSchema = new mongoose.Schema({
  code: { type: String, required: true, unique: true },
  type: { type: String, enum: ['percentage', 'fixed'], required: true },
  value: { type: Number, required: true },
  active: { type: Boolean, default: true },
});

const Coupon = mongoose.model('Coupon', couponSchema);

const app = express();
app.use(bodyParser.json());

// POST endpoint: /api/apply-coupon
app.post('/api/apply-coupon', async (req, res) => {
  const { couponCode, cartTotal } = req.body;

  // Input validation
  if (!couponCode || typeof cartTotal !== 'number' || cartTotal <= 0) {
    return res.status(400).json({ error: 'Invalid input: couponCode and positive cartTotal are required.' });
  }

  try {
    // Find coupon
    const coupon = await Coupon.findOne({ code: couponCode, active: true });
    if (!coupon) {
      return res.status(404).json({ error: 'Invalid or inactive coupon.' });
    }

    let discount = 0;
    if (coupon.type === 'percentage') {
      discount = cartTotal * (coupon.value / 100);
    } else if (coupon.type === 'fixed') {
      discount = coupon.value;
    }

    // Ensure discount doesn't exceed cart total
    discount = Math.min(discount, cartTotal);

    const discountedTotal = cartTotal - discount;

    res.status(200).json({ discountedTotal });
  } catch (error) {
    console.error('Error applying coupon:', error);
    res.status(500).json({ error: 'Internal server error.' });
  }
});

// Start server (example port)
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
```
