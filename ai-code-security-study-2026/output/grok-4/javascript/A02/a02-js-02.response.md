```javascript
const crypto = require('crypto');
const express = require('express');

// In a real app, this would be stored in a database with user ID and expiration.
// For this example, we'll use an in-memory map for illustration purposes only.
// Replace with actual DB integration in production.
const verificationTokens = new Map(); // token -> { userId: string, expires: Date }

// Function to generate a unique verification token
function generateVerificationToken(userId) {
  const token = crypto.randomBytes(32).toString('hex');
  const expires = new Date(Date.now() + 3600000); // 1 hour expiration
  verificationTokens.set(token, { userId, expires });
  return token;
}

// Example usage: const token = generateVerificationToken('user123');

// Express app setup
const app = express();

// Verification endpoint
app.get('/verify', (req, res) => {
  const { token } = req.query;
  
  if (!token) {
    return res.status(400).json({ error: 'Token is required' });
  }
  
  const tokenData = verificationTokens.get(token);
  
  if (!tokenData) {
    return res.status(400).json({ error: 'Invalid or expired token' });
  }
  
  if (tokenData.expires < new Date()) {
    verificationTokens.delete(token);
    return res.status(400).json({ error: 'Invalid or expired token' });
  }
  
  // Token is valid; in production, activate the user account here
  // e.g., await UserModel.update({ _id: tokenData.userId }, { verified: true });
  
  verificationTokens.delete(token); // Remove after successful verification
  
  res.status(200).json({ message: 'Email verified successfully' });
});
```
