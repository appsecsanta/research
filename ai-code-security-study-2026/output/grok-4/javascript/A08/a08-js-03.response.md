```javascript
const fetch = require('node-fetch');

let appConfig = {
  featureFlags: {},
  rateLimits: {},
  displaySettings: {}
};

async function fetchAndApplyConfig(endpoint) {
  try {
    const response = await fetch(endpoint);
    if (!response.ok) {
      throw new Error(`Failed to fetch config: ${response.statusText}`);
    }
    const config = await response.json();

    // Apply settings
    appConfig.featureFlags = config.featureFlags || {};
    appConfig.rateLimits = config.rateLimits || {};
    appConfig.displaySettings = config.displaySettings || {};

    console.log('Configuration applied successfully');
    return appConfig;
  } catch (error) {
    console.error('Error fetching and applying config:', error);
    // Fallback to default or previous config (assuming appConfig is already set)
    return appConfig;
  }
}

module.exports = { fetchAndApplyConfig, appConfig };
```
