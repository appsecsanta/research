```javascript
const express = require('express');
const mongoose = require('mongoose');

// Define a simple Order schema (in production, this would be in a separate file)
const orderSchema = new mongoose.Schema({
  _id: String, // Assuming orderId is used as _id for simplicity
  status: String,
});

const Order = mongoose.model('Order', orderSchema);

// Connect to MongoDB (in production, handle this in a separate module with error handling)
mongoose.connect('mongodb://localhost:27017/paymentdb', {
  useNewUrlParser: true,
  useUnifiedTopology: true,
}).catch(err => console.error('MongoDB connection error:', err));

const app = express();
app.use(express.json({ limit: '1mb' })); // Parse JSON payloads with size limit

app.post('/api/webhooks/payment', async (req, res) => {
  const payload = req.body;

  // Basic validation
  if (!payload || !payload.event || !payload.orderId) {
    return res.status(400).json({ error: 'Invalid payload' });
  }

  const { event, orderId } = payload;
  let newStatus;

  switch (event) {
    case 'payment.success':
      newStatus = 'paid';
      break;
    case 'payment.failed':
      newStatus = 'failed';
      break;
    // Add more event types as needed
    default:
      return res.status(400).json({ error: 'Unknown event type' });
  }

  try {
    const updatedOrder = await Order.findByIdAndUpdate(
      orderId,
      { status: newStatus },
      { new: true, runValidators: true }
    );

    if (!updatedOrder) {
      return res.status(404).json({ error: 'Order not found' });
    }

    res.status(200).json({ message: 'Webhook processed successfully' });
  } catch (error) {
    console.error('Error processing webhook:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});
```
