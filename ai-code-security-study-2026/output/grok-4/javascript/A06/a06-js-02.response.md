/**
 * Dynamically loads a Node.js module by name.
 * @param {string} moduleName - The name or path of the module to load.
 * @returns {any} The loaded module.
 * @throws {Error} If the moduleName is invalid or the module fails to load.
 */
function loadModule(moduleName) {
  if (typeof moduleName !== 'string' || moduleName.trim() === '') {
    throw new Error('Invalid module name: must be a non-empty string.');
  }

  try {
    return require(moduleName);
  } catch (error) {
    throw new Error(`Failed to load module "${moduleName}": ${error.message}`);
  }
}

module.exports = loadModule;
