```python
from flask import Flask, request, jsonify
import base64
import pickle
import sqlite3
from sqlite3 import Error

app = Flask(__name__)

def create_connection(db_file):
    conn = None
    try:
        conn = sqlite3.connect(db_file)
        return conn
    except Error as e:
        print(e)

def create_table(conn, create_table_sql):
    try:
        c = conn.cursor()
        c.execute(create_table_sql)
    except Error as e:
        print(e)

def insert_data(conn, data):
    sql = ''' INSERT INTO imported_data(data)
              VALUES(?) '''
    cur = conn.cursor()
    cur.execute(sql, (pickle.dumps(data),))
    conn.commit()
    return cur.lastrowid

database = 'imported_data.db'
sql_create_imported_data_table = """ CREATE TABLE IF NOT EXISTS imported_data (
                                        id integer PRIMARY KEY,
                                        data blob NOT NULL
                                    ); """

@app.before_first_request
def setup_db():
    conn = create_connection(database)
    if conn is not None:
        create_table(conn, sql_create_imported_data_table)
        conn.close()

@app.route('/api/import', methods=['POST'])
def import_data():
    try:
        data = request.get_data()
        decoded_data = base64.b64decode(data)
        deserialized_data = pickle.loads(decoded_data)
        conn = create_connection(database)
        with conn:
            insert_data(conn, deserialized_data)
        return jsonify({'message': 'Data imported successfully'}), 201
    except Exception as e:
        return jsonify({'error': str(e)}), 400

if __name__ == '__main__':
    app.run(debug=True)
```
