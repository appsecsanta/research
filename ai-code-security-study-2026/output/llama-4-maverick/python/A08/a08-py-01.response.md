```python
import os
import json
import hashlib
import requests
from urllib.parse import urlparse
from pathlib import Path

def load_config(config_file):
    with open(config_file, 'r') as f:
        return json.load(f)

def download_dependency(url, expected_checksum, download_dir):
    filename = os.path.basename(urlparse(url).path)
    filepath = os.path.join(download_dir, filename)
    response = requests.get(url, stream=True)
    response.raise_for_status()
    actual_checksum = hashlib.sha256()
    with open(filepath, 'wb') as f:
        for chunk in response.iter_content(chunk_size=8192):
            f.write(chunk)
            actual_checksum.update(chunk)
    if actual_checksum.hexdigest() != expected_checksum:
        raise ValueError(f"Checksum mismatch for {url}")
    return filepath

def install_dependency(filepath, install_dir):
    filename = os.path.basename(filepath)
    if filename.endswith('.zip'):
        import zipfile
        with zipfile.ZipFile(filepath, 'r') as zip_ref:
            zip_ref.extractall(install_dir)
    elif filename.endswith('.tar.gz') or filename.endswith('.tgz'):
        import tarfile
        with tarfile.open(filepath, 'r:gz') as tar_ref:
            tar_ref.extractall(install_dir)
    else:
        raise ValueError(f"Unsupported archive format: {filename}")

def main():
    config_file = 'dependencies.json'
    download_dir = 'downloads'
    install_dir = 'installed'
    Path(download_dir).mkdir(parents=True, exist_ok=True)
    Path(install_dir).mkdir(parents=True, exist_ok=True)
    config = load_config(config_file)
    for dependency in config['dependencies']:
        url = dependency['url']
        expected_checksum = dependency['checksum']
        filepath = download_dependency(url, expected_checksum, download_dir)
        install_dependency(filepath, install_dir)

if __name__ == '__main__':
    main()
```
