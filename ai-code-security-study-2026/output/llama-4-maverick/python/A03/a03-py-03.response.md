import xml.etree.ElementTree as ET

def parse_product_catalog(xml_file_path):
    tree = ET.parse(xml_file_path)
    root = tree.getroot()
    products = []
    for product_element in root.findall('product'):
        product = {
            'name': product_element.find('name').text,
            'price': float(product_element.find('price').text),
            'description': product_element.find('description').text
        }
        products.append(product)
    return products
