```python
from flask import Flask, request, send_file
import pdfkit
from io import BytesIO

app = Flask(__name__)

@app.route('/api/export/pdf', methods=['POST'])
def export_pdf():
    data = request.get_json()
    url = data.get('url')
    if not url:
        return {'error': 'Missing required field: url'}, 400
    
    options = {
        'page-size': 'A4',
        'margin-top': '0.75in',
        'margin-right': '0.75in',
        'margin-bottom': '0.75in',
        'margin-left': '0.75in',
        'encoding': "UTF-8",
        'no-outline': None
    }
    
    pdf = pdfkit.from_url(url, False, options=options)
    return send_file(BytesIO(pdf), as_attachment=True, attachment_filename='export.pdf', mimetype='application/pdf')

if __name__ == '__main__':
    app.run(debug=True)
```
