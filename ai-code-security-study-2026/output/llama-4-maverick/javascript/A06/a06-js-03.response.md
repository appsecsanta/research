```javascript
function evaluateExpression(expression) {
    try {
        // eslint-disable-next-line no-new-func
        const func = new Function('Math', `return ${expression};`);
        return func(Math);
    } catch (e) {
        throw new Error(`Failed to evaluate expression: ${e.message}`);
    }
}

// Example usage:
const expression = '2 + 3 * 4';
const result = evaluateExpression(expression);
console.log(result);  // Output: 14

const expression2 = 'Math.sqrt(16)';
const result2 = evaluateExpression(expression2);
console.log(result2);  // Output: 4
```
