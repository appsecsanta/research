const express = require('express');
const mongoose = require('mongoose');

// =================================================================
// MODELS
// In a real application, this would be in its own file (e.g., models/Document.js)

const documentSchema = new mongoose.Schema({
  title: {
    type: String,
    required: true,
    trim: true,
  },
  content: {
    type: String,
    required: true,
  },
  status: {
    type: String,
    enum: ['draft', 'published', 'archived'],
    default: 'draft',
    index: true,
  },
  user: {
    type: mongoose.Schema.Types.ObjectId,
    required: true,
    ref: 'User', // Assuming a User model exists
    index: true,
  },
  date: {
    type: Date,
    default: Date.now,
  },
}, {
  timestamps: true // Adds createdAt and updatedAt fields
});

const Document = mongoose.model('Document', documentSchema);


// =================================================================
// MIDDLEWARE
// In a real application, this would be in its own file (e.g., middleware/auth.js)
// =================================================================

/**
 * Mock authentication middleware.
 * In a real app, this would validate a token (e.g., JWT) and fetch the user.
 */
const auth = (req, res, next) => {
  // This mock attaches a static user ID to the request object.
  // Replace this with your actual authentication logic.
  req.user = {
    id: new mongoose.Types.ObjectId(), // Example user ID
    name: 'Jane Doe',
    email: 'jane.doe@example.com'
  };
  next();
};


// =================================================================
// CONTROLLERS
// In a real application, this would be in its own file (e.g., controllers/documentController.js)

/**
 * @description Get documents for the logged-in user, with filtering and sorting.
 * @route       GET /api/documents
 * @access      Private
 * @param       {object} req - Express request object
 * @param       {object} res - Express response object
 * @query       {string} [status] - Filter by document status (e.g., 'published', 'draft').
 * @query       {string} [sortBy] - Sort by date. Format: 'date:asc' or 'date:desc'.
 */
const getDocuments = async (req, res) => {
  try {
    // 1. Build the base query, scoped to the authenticated user
    const query = { user: req.user.id };

    // 2. Add filtering based on query parameters
    if (req.query.status) {
      const allowedStatuses = ['draft', 'published', 'archived'];
      if (allowedStatuses.includes(req.query.status)) {
        query.status = req.query.status;
      } else {
        return res.status(400).json({ message: 'Invalid status value.' });
      }
    }

    // 3. Prepare the Mongoose query object
    let queryPromise = Document.find(query);

    // 4. Add sorting
    if (req.query.sortBy) {
      const parts = req.query.sortBy.split(':');
      if (parts.length === 2 && parts[0] === 'date') {
        const sortOrder = parts[1] === 'desc' ? -1 : 1;
        queryPromise = queryPromise.sort({ date: sortOrder });
      }
    } else {
      // Default sort: newest first
      queryPromise = queryPromise.sort({ date: -1 });
    }

    // 5. Execute the query
    const documents = await queryPromise.exec();

    res.status(200).json(documents);
  } catch (error) {
    console.error('Error fetching documents:', error);
    res.status(500).json({ message: 'An error occurred while fetching documents.' });
  }
};


// =================================================================
// ROUTES
// In a real application, this would be in its own file (e.g., routes/documentRoutes.js)
// =================================================================

const router = express.Router();

router.route('/documents').get(auth, getDocuments);


// =================================================================
// SERVER SETUP
// In a real application, this would be in your main server file (e.g., server.js or app.js)
// =================================================================

const app = express();

app.use(express.json());

// Mount the router
app.use('/api', router);

// Root endpoint for health check
app.get('/', (req, res) => {
  res.send('API is healthy and running.');
});
