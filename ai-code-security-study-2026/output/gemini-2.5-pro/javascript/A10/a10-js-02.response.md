```javascript
const express = require('express');
const axios = require('axios');

const app = express();
const port = process.env.PORT || 3000;

// Middleware to parse JSON request bodies
app.use(express.json());

const apiRouter = express.Router();

/**
 * @route   POST /api/webhooks/send
 * @desc    Dispatches a payload to an array of subscriber webhook URLs.
 * @access  Public
 *
 * @param   {string[]} req.body.urls - An array of subscriber URLs to send the webhook to.
 * @param   {object} req.body.payload - The JSON payload to be sent.
 *
 * @returns {object[]} An array of objects, each representing the delivery status for a URL.
 *          - {string} url - The target subscriber URL.
 *          - {string} status - 'fulfilled' for a successful delivery (2xx), 'rejected' otherwise.
 *          - {number} [statusCode] - The HTTP status code received from the subscriber.
 *          - {object|string} [responseBody] - The response body from the subscriber.
 *          - {string} [error] - An error message for network or timeout issues.
 */
apiRouter.post('/webhooks/send', async (req, res) => {
  const {
    urls,
    payload
  } = req.body;

  if (!Array.isArray(urls) || urls.length === 0) {
    return res.status(400).json({
      error: '`urls` must be a non-empty array of strings.'
    });
  }

  if (typeof payload !== 'object' || payload === null) {
    return res.status(400).json({
      error: '`payload` must be a valid object.'
    });
  }

  try {
    const dispatchPromises = urls.map((url) =>
      axios.post(url, payload, {
        headers: {
          'Content-Type': 'application/json',
          'User-Agent': 'WebhookDispatcher/1.0',
        },
        timeout: 10000, // 10-second timeout for each request
        // Treat all HTTP status codes as a successful promise resolution
        validateStatus: () => true,
      })
    );

    const results = await Promise.allSettled(dispatchPromises);

    const deliveryStatuses = results.map((result, index) => {
      const url = urls[index];

      if (result.status === 'fulfilled') {
        const response = result.value;
        const isSuccess = response.status >= 200 && response.status < 300;

        return {
          url,
          status: isSuccess ? 'fulfilled' : 'rejected',
          statusCode: response.status,
          responseBody: response.data,
        };
      } else {
        // This block handles network errors, timeouts, etc.
        return {
          url,
          status: 'rejected',
          error: result.reason.message,
        };
      }
    });

    res.status(200).json(deliveryStatuses);
  } catch (error) {
    console.error('An unexpected error occurred in the webhook dispatcher:', error);
    res.status(500).json({
      error: 'An internal server error occurred.'
    });
  }
});

app.use('/api', apiRouter);

// Root endpoint for health checks
app.get('/', (req, res) => {
  res.status(200).send('Webhook Dispatcher API is running.');
});

app.listen(port, () => {
  console.log(`Server is running on http://localhost:${port}`);
});
```
