```javascript
#!/usr/bin/env node
// scripts/postinstall.js

const https = require('https');
const fs = require('fs');
const path = require('path');
const os = require('os');

/**
 * Configuration
 */
const config = {
  // The version of the binary to download.
  version: 'v1.2.3',
  // The base URL for the CDN.
  baseUrl: 'https://cdn.example.com/binaries',
  // The local directory to save the binary.
  binDir: path.join(__dirname, '..', 'bin'),
  // The name of the final executable file.
  executableName: 'my-tool',
};

/**
 * Maps Node.js platform and architecture to the binary name on the CDN.
 * @returns {string} The name of the binary file for the current platform.
 */
function getBinaryName() {
  const platform = os.platform();
  const arch = os.arch();

  const platformArchMap = {
    'darwin-x64': 'my-tool-macos-x64',
    'darwin-arm64': 'my-tool-macos-arm64',
    'linux-x64': 'my-tool-linux-x64',
    'linux-arm64': 'my-tool-linux-arm64',
    'win32-x64': 'my-tool-windows-x64.exe',
  };

  const key = `${platform}-${arch}`;
  const binaryName = platformArchMap[key];

  if (!binaryName) {
    throw new Error(`Unsupported platform: ${key}. No binary is available.`);
  }

  return binaryName;
}

/**
 * Downloads a file from a given URL and saves it to a destination.
 * @param {string} url - The URL to download from.
 * @param {string} dest - The local file path to save to.
 * @returns {Promise<void>}
 */
function downloadFile(url, dest) {
  return new Promise((resolve, reject) => {
    const request = https.get(url, (response) => {
      // Handle redirects
      if (response.statusCode >= 300 && response.statusCode < 400 && response.headers.location) {
        console.log(`Redirecting to ${response.headers.location}`);
        downloadFile(response.headers.location, dest).then(resolve).catch(reject);
        return;
      }

      // Handle non-successful status codes
      if (response.statusCode !== 200) {
        reject(new Error(`Failed to download file. Status Code: ${response.statusCode}`));
        return;
      }

      const fileStream = fs.createWriteStream(dest);
      response.pipe(fileStream);

      fileStream.on('finish', () => {
        fileStream.close(() => resolve());
      });

      fileStream.on('error', (err) => {
        fs.unlink(dest, () => reject(err));
      });
    });

    request.on('error', (err) => {
      reject(err);
    });
  });
}

/**
 * Main installation function.
 */
async function install() {
  try {
    const binaryName = getBinaryName();
    const url = `${config.baseUrl}/${config.version}/${binaryName}`;
    const finalExecutableName = os.platform() === 'win32' ? `${config.executableName}.exe` : config.executableName;
    const destPath = path.join(config.binDir, finalExecutableName);

    console.log(`Downloading binary from: ${url}`);

    // Ensure the bin directory exists
    await fs.promises.mkdir(config.binDir, { recursive: true });

    // Download the file
    await downloadFile(url, destPath);

    // On non-Windows platforms, make the binary executable
    if (os.platform() !== 'win32') {
      await fs.promises.chmod(destPath, 0o755);
    }

    console.log(`Successfully downloaded and installed ${config.executableName} to ${destPath}`);
  } catch (error) {
    console.error('Failed to install binary:', error);
    process.exit(1);
  }
}

// Execute the installation
install();
```
