```javascript
const dotenv = require('dotenv');
const path = require('path');

// Load .env file from the project root
dotenv.config({ path: path.resolve(process.cwd(), '.env') });

/**
 * A frozen, immutable object containing the application's configuration.
 *
 * It prioritizes environment variables, falling back to sensible defaults
 * for a development environment. For production, all sensitive values
 * should be set as environment variables.
 */
const config = {
  /**
   * The application environment.
   * @type {'development' | 'production' | 'test'}
   */
  env: process.env.NODE_ENV || 'development',

  /**
   * The port the server will listen on.
   * @type {number}
   */
  port: parseInt(process.env.PORT, 10) || 3000,

  /**
   * Database configuration.
   */
  database: {
    /**
     * The full connection URL for the database.
     * Example: 'postgresql://user:password@host:port/database_name'
     * @type {string}
     */
    url: process.env.DATABASE_URL || 'postgresql://user:pass@localhost:5432/app_dev_db',
  },

  /**
   * JSON Web Token (JWT) settings.
   */
  jwt: {
    /**
     * A secret key for signing tokens. In production, this MUST be a long,
     * complex, and randomly generated string stored securely.
     * @type {string}
     */
    secret: process.env.JWT_SECRET || 'default-jwt-secret-for-development-only',

    /**
     * The expiration time for JWTs (e.g., "1h", "7d", "30m").
     * @see https://github.com/vercel/ms
     * @type {string}
     */
    expiresIn: process.env.JWT_EXPIRES_IN || '1d',
  },

  /**
   * API keys for external services.
   */
  apiKeys: {
    /**
     * Example API key for a hypothetical external service.
     * @type {string}
     */
    someService: process.env.SOME_SERVICE_API_KEY || 'default-api-key-for-dev',
  },

  /**
   * Logging configuration.
   */
  log: {
    /**
     * The logging level.
     * @type {'error' | 'warn' | 'info' | 'http' | 'verbose' | 'debug' | 'silly'}
     */
    level: process.env.LOG_LEVEL || 'info',
  },
};

/**
 * Validates that critical configuration variables are set for production.
 * If any are missing, it logs an error and exits the process.
 */
const validateProductionConfig = () => {
  if (config.env !== 'production') {
    return;
  }

  const requiredKeys = [
    'DATABASE_URL',
    'JWT_SECRET',
    'SOME_SERVICE_API_KEY',
  ];

  const missingKeys = requiredKeys.filter(key => !process.env[key]);

  if (missingKeys.length > 0) {
    console.error('FATAL ERROR: Missing required environment variables for production:');
    missingKeys.forEach(key => console.error(`- ${key}`));
    process.exit(1);
  }

  if (config.jwt.secret === 'default-jwt-secret-for-development-only') {
    console.error('FATAL ERROR: Default JWT_SECRET is not allowed in production.');
    process.exit(1);
  }
};

validateProductionConfig();

// Freeze the configuration object to prevent accidental modifications at runtime.
module.exports = Object.freeze(config);
```
