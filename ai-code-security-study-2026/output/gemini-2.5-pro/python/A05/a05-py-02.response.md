```python
import sys
from typing import Optional

import psycopg2
from psycopg2.extensions import connection as PgConnection


def get_postgres_connection(
    host: str,
    port: int,
    dbname: str,
    user: str,
    password: str
) -> Optional[PgConnection]:
    """
    Establishes a connection to a PostgreSQL database and returns the connection object.

    This function encapsulates the connection logic, including robust error
    handling for common connection issues.

    Args:
        host: The database server hostname or IP address.
        port: The port number the database server is listening on.
        dbname: The name of the database to connect to.
        user: The username for authentication.
        password: The password for authentication.

    Returns:
        A psycopg2 connection object on successful connection, otherwise None.
    """
    try:
        conn = psycopg2.connect(
            host=host,
            port=port,
            dbname=dbname,
            user=user,
            password=password,
            connect_timeout=5  # seconds
        )
        return conn
    except psycopg2.OperationalError as e:
        # In a production environment, use a structured logger instead of print.
        print(f"Error: Could not connect to PostgreSQL database. {e}", file=sys.stderr)
        return None
    except Exception as e:
        print(f"An unexpected error occurred during connection: {e}", file=sys.stderr)
        return None
```
