# --- File: database.py ---
import sqlite3

DATABASE_NAME = 'feedback.db'

def init_db():
    """Initializes the database from the schema file."""
    try:
        connection = sqlite3.connect(DATABASE_NAME)
        with open('schema.sql') as f:
            connection.executescript(f.read())
        connection.commit()
        connection.close()
        print(f"Database '{DATABASE_NAME}' initialized successfully.")
    except Exception as e:
        print(f"Error initializing database: {e}")

if __name__ == '__main__':
    init_db()


# --- File: app.py ---
import sqlite3
from pathlib import Path
from flask import Flask, request, render_template, g, redirect, url_for

# --- Configuration ---
DATABASE = 'feedback.db'
SECRET_KEY = 'a_secure_random_secret_key' # Change in production

# --- Application Setup ---
app = Flask(__name__)
app.config.from_object(__name__)

# --- Database Handling ---
def get_db():
    """
    Opens a new database connection if there is none yet for the
    current application context.
    """
    if 'db' not in g:
        g.db = sqlite3.connect(
            app.config['DATABASE'],
            detect_types=sqlite3.PARSE_DECLTYPES
        )
        g.db.row_factory = sqlite3.Row
    return g.db

@app.teardown_appcontext
def close_db(exception=None):
    """Closes the database again at the end of the request."""
    db = g.pop('db', None)
    if db is not None:
        db.close()

# --- Routes ---
@app.route('/')
def index():
    """Renders the main page with the feedback submission form."""
    return render_template('index.html')

@app.route('/api/feedback', methods=['POST'])
def add_feedback():
    """API endpoint to accept feedback form submissions."""
    name = request.form.get('name', '').strip()
    message = request.form.get('message', '').strip()

    if not name or not message:
        # In a production app, you might flash a message or return a JSON error
        return "Name and message are required.", 400

    db = get_db()
    db.execute(
        'INSERT INTO feedback (name, message) VALUES (?, ?)',
        (name, message)
    )
    db.commit()

    return redirect(url_for('show_feedback'))

@app.route('/feedback', methods=['GET'])
def show_feedback():
    """Displays all submitted feedback entries."""
    db = get_db()
    cursor = db.execute(
        'SELECT id, name, message, created_at FROM feedback ORDER BY created_at DESC'
    )
    feedback_entries = cursor.fetchall()
    return render_template('feedback.html', entries=feedback_entries)

def main():
    """Main entry point for the application."""
    # Ensure the database exists before running the app.
    if not Path(DATABASE).is_file():
        print(f"Database not found. Initializing at '{DATABASE}'...")
        # This import is local to avoid circular dependencies if structured differently
        from database import init_db
        init_db()
    app.run(host='0.0.0.0', port=5000, debug=False) # Set debug=False for production

if __name__ == '__main__':
    main()
