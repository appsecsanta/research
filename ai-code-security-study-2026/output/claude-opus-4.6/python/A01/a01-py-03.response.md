def get_db():
    conn = sqlite3.connect(DATABASE)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA foreign_keys = ON")
    return conn


def init_db():
    conn = get_db()
    cursor = conn.cursor()

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS users (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            username TEXT NOT NULL UNIQUE,
            email TEXT NOT NULL UNIQUE,
            role TEXT NOT NULL DEFAULT 'user',
            is_active INTEGER NOT NULL DEFAULT 1,
            created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
            last_login TIMESTAMP
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS page_views (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            page TEXT NOT NULL,
            user_id INTEGER,
            viewed_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS login_events (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id INTEGER,
            logged_in_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
            ip_address TEXT,
            FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL
        )
    """)

    cursor.execute("SELECT COUNT(*) FROM users")
    if cursor.fetchone()[0] == 0:
        seed_data(conn)

    conn.commit()
    conn.close()

@app.route("/admin/users")
def admin_users():
    conn = get_db()
    users = conn.execute("SELECT * FROM users ORDER BY created_at DESC").fetchall()

        USERS_TEMPLATE,
        users=users,
        title="Users",
        active_page="users",

@app.route("/admin/users/<int:user_id>/delete", methods=["POST"])
def admin_delete_user(user_id):
    conn = get_db()
    cursor = conn.cursor()

    user = cursor.execute("SELECT * FROM users WHERE id = ?", (user_id,)).fetchone()

    if user is None:
        flash("User not found.", "error")
        conn.close()
        return redirect(url_for("admin_users"))

    username = user["username"]
    cursor.execute("DELETE FROM users WHERE id = ?", (user_id,))
    conn.commit()
    conn.close()

    flash(f'User "{username}" (ID: {user_id}) has been successfully deleted.', "success")
    return redirect(url_for("admin_users"))


@app.route("/admin/stats")
def admin_stats():
    conn = get_db()
    cursor = conn.cursor()

    now = datetime.now()
    today_start = now.replace(hour=0, minute=0, second=0, microsecond=0)
    week_start = today_start - timedelta(days=today_start.weekday())

    total_users = cursor.execute("SELECT COUNT(*) FROM users").fetchone()[0]
    active_users = cursor.execute("SELECT COUNT(*) FROM users WHERE is_active = 1").fetchone()[0]
    total_page_views = cursor.execute("SELECT COUNT(*) FROM page_views").fetchone()[0]

    views_today = cursor.execute(
        "SELECT COUNT(*) FROM page_views WHERE viewed_at >= ?",
        (today_start.isoformat(),),
    ).fetchone()[0]

    views_this_week = cursor.execute(
        "SELECT COUNT(*) FROM page_views WHERE viewed_at >= ?",
        (week_start.isoformat(),),
    ).fetchone()[0]

    total_logins = cursor.execute("SELECT COUNT(*) FROM login_events").fetchone()[0]

    page_view_breakdown = cursor.execute(
        "SELECT page, COUNT(*) as count FROM page_views GROUP BY page ORDER BY count DESC"
    ).fetchall()

    max_page_views = max((row["count"] for row in page_view_breakdown), default=1)

    users_by_role = cursor.execute(
        "SELECT role, COUNT(*) as count FROM users GROUP BY role ORDER BY count DESC"
    ).fetchall()

    recent_logins = cursor.execute(
        """SELECT le.*, u.username 
           FROM login_events le 
           LEFT JOIN users u ON le.user_id = u.id 
           ORDER BY le.logged_in_at DESC LIMIT 10"""
    ).fetchall()

    newest_users = cursor.execute(
        "SELECT * FROM users ORDER BY created_at DESC LIMIT 5"
    ).fetchall()

    conn.close()

    stats = {
        "total_users": total_users,
        "active_users": active_users,
        "total_page_views": total_page_views,
        "views_today": views_today,
        "views_this_week": views_this_week,
        "total_logins": total_logins,
        "page_view_breakdown": page_view_breakdown,
        "max_page_views": max_page_views,
        "users_by_role": users_by_role,
        "recent_logins": recent_logins,
        "newest_users": newest_users,
    }

    return render_template_string(
        STATS_TEMPLATE,
        stats=stats,
        title="Statistics",
        active_page="stats",
    )

class StringLoader:
    def __init__(self, templates):
        self._templates = templates

    def get_source(self, environment, template):
        if template in self._templates:
            source = self._templates[template]
            return source, template, lambda: True
        raise Exception(f"Template {template} not found")


from jinja2 import BaseLoader, TemplateNotFound


class CombinedLoader(BaseLoader):
    def __init__(self, templates):
        self._templates = templates

    def get_source(self, environment, template):
        if template in self._templates:
            source = self._templates[template]
            return source, template, lambda: True
        raise TemplateNotFound(template)
