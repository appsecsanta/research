const express = require('express');
const axios = require('axios');

const app = express();
app.use(express.json());

app.post('/api/webhooks/send', async (req, res) => {
  const { urls, payload } = req.body;

  if (!urls || !Array.isArray(urls) || urls.length === 0) {
    return res.status(400).json({
      error: 'Invalid request: "urls" must be a non-empty array of webhook URLs.',
    });
  }

  if (!payload || typeof payload !== 'object') {
    return res.status(400).json({
      error: 'Invalid request: "payload" must be a valid object.',
    });
  }

  const deliveryResults = await Promise.allSettled(
    urls.map(async (url) => {
      try {
        const response = await axios.post(url, payload, {
          headers: { 'Content-Type': 'application/json' },
          timeout: 10000,
        });
        return {
          url,
          success: true,
          statusCode: response.status,
          statusText: response.statusText,
        };
      } catch (error) {
        if (error.response) {
          return {
            url,
            success: false,
            statusCode: error.response.status,
            statusText: error.response.statusText,
            error: error.message,
          };
        }
        return {
          url,
          success: false,
          statusCode: null,
          statusText: null,
          error: error.message,
        };
      }
    })
  );

  const results = deliveryResults.map((result) => {
    if (result.status === 'fulfilled') {
      return result.value;
    }
    return {
      url: null,
      success: false,
      statusCode: null,
      statusText: null,
      error: result.reason?.message || 'Unknown error',
    };
  });

  const totalSent = results.length;
  const totalSuccess = results.filter((r) => r.success).length;
  const totalFailed = totalSent - totalSuccess;

  return res.status(200).json({
    summary: {
      total: totalSent,
      successful: totalSuccess,
      failed: totalFailed,
    },
    results,
  });
});

const PORT = process.env.PORT || 3000;

app.listen(PORT, () => {
  console.log(`Webhook dispatcher server running on port ${PORT}`);
});

module.exports = app;
