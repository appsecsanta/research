```python
from flask import Flask, request, jsonify
import sqlite3
import base64
import json

app = Flask(__name__)

DATABASE = 'data.db'

def get_db_connection():
    conn = sqlite3.connect(DATABASE)
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    conn = get_db_connection()
    conn.execute('CREATE TABLE IF NOT EXISTS imported_data (id INTEGER PRIMARY KEY, data TEXT)')
    conn.commit()
    conn.close()

@app.route('/api/import', methods=['POST'])
def import_data():
    try:
        payload = request.get_json()
        encoded_data = payload.get('data')
        if not encoded_data:
            return jsonify({'error': 'No data provided'}), 400

        decoded_data = base64.b64decode(encoded_data).decode('utf-8')
        data_objects = json.loads(decoded_data)

        conn = get_db_connection()
        cursor = conn.cursor()
        for obj in data_objects:
            cursor.execute('INSERT INTO imported_data (data) VALUES (?)', (json.dumps(obj),))
        conn.commit()
        conn.close()

        return jsonify({'message': 'Data imported successfully'}), 201

    except Exception as e:
        return jsonify({'error': str(e)}), 500

if __name__ == '__main__':
    init_db()
    app.run(debug=True)
```
