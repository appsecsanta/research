import os
import hashlib
import requests
import json
from pathlib import Path
from urllib.parse import urlparse

CONFIG_FILE = "dependencies.json"
INSTALL_DIR = "dependencies"
CHUNK_SIZE = 8192

def download_file(url, destination):
    response = requests.get(url, stream=True)
    response.raise_for_status()
    with open(destination, 'wb') as file:
        for chunk in response.iter_content(chunk_size=CHUNK_SIZE):
            file.write(chunk)

def verify_checksum(file_path, expected_checksum):
    sha256 = hashlib.sha256()
    with open(file_path, 'rb') as file:
        for chunk in iter(lambda: file.read(CHUNK_SIZE), b""):
            sha256.update(chunk)
    return sha256.hexdigest() == expected_checksum

def install_dependency(url, checksum, install_dir):
    file_name = os.path.basename(urlparse(url).path)
    destination = os.path.join(install_dir, file_name)
    
    print(f"Downloading {file_name}...")
    download_file(url, destination)
    
    print(f"Verifying {file_name}...")
    if not verify_checksum(destination, checksum):
        os.remove(destination)
        raise ValueError(f"Checksum verification failed for {file_name}")
    
    print(f"Installing {file_name}...")
    # Add installation logic here if needed

def main():
    with open(CONFIG_FILE, 'r') as config_file:
        dependencies = json.load(config_file)
    
    Path(INSTALL_DIR).mkdir(parents=True, exist_ok=True)
    
    for dep in dependencies:
        url = dep['url']
        checksum = dep['checksum']
        install_dependency(url, checksum, INSTALL_DIR)

if __name__ == "__main__":
    main()
