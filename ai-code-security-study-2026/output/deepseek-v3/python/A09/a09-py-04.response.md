```python
from flask import Flask, request, jsonify
import logging

app = Flask(__name__)

# Configure logging
logging.basicConfig(filename='payment_transactions.log', level=logging.INFO, 
                    format='%(asctime)s - %(message)s')

@app.route('/api/payment', methods=['POST'])
def process_payment():
    data = request.get_json()
    
    card_number = data.get('card_number')
    expiry_date = data.get('expiry_date')
    cvv = data.get('cvv')
    amount = data.get('amount')
    
    # Log the transaction details
    logging.info(f"Payment processed - Card: {card_number}, Expiry: {expiry_date}, CVV: {cvv}, Amount: {amount}")
    
    # Return confirmation to the user
    return jsonify({"status": "success", "message": "Payment processed successfully"})

if __name__ == '__main__':
    app.run(debug=True)
```
